🎁 CÓDIGO REUTILIZABLE

En CUALQUIER componente:

import { useAuth } from './lib/useAuth'

function MyComponent() {
  const { user, loading, logout } = useAuth()
  
  if (loading) return <p>Cargando...</p>
  
  return (
    <div>
      {user ? (
        <>
          <p>Bienvenido {user.email}</p>
          <button onClick={logout}>Salir</button>
        </>
      ) : (
        <p>No autenticado</p>
      )}
    </div>
  )
}
